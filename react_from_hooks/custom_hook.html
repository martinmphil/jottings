Custom hooks

By convention, names for custom hooks begin with the letters "use", eg "useWindowWidth()". Starting custom hook names with "use" allows react linters to debug hook functions.

In the following example, any other function component can call "useWindowWidth()" as a hook.

function useWindowWidth() {
   const [width, setWidth] = useState(window.innerWidth)
   useEffect(() => {
     const handleResize = () => setWidth(window.innerWidth)
     window.addEventListener('resize', handleResize)
     return () => {
       window.removeEventListener('resize', handleResize)
     }
   })
   return width
} 
    
// Code example

import React, { useState } from 'react'

function MyComponent() {

  const width = useWidth()

  return (
    <div>
      //component utilising {width}
    </div>
  )
}
