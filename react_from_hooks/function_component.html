Function components

Function component accept props as an argument and return html elements inside a single html element.

NB "props" is short for short for properties. 

    
const my_component = (props) => {
  ...
  return <div />
}
    
  
    
function my_component(props) {
  ...
  return <div />
}
    
  
    
// Code example

function Welcome(props) {
  return <h1>Hello, {props.name}</h1>
}
